//CREAR UNA TABLA DE USUARIOS 
CREATE TABLE Usuarios (
    IDUsuario SERIAL PRIMARY KEY,
    IDPersona INT,
    NombreUsuario VARCHAR(50) NOT NULL,
    Contraseña VARCHAR(255) NOT NULL, -- Se recomienda almacenar hashes de contraseñas en lugar de contraseñas en texto plano
    Nivel INT,
    Estado VARCHAR(20),
    FOREIGN KEY (IDPersona) REFERENCES Persona(id)
);
